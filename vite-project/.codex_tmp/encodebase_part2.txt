      return normalized <= 0.03928
        ? normalized / 12.92
        : ((normalized + 0.055) / 1.055) ** 2.4

          variant === 'burst'
            ? 24 + Math.random() * 38
            : variant === 'trail'

          color: palette[Math.floor(Math.random() * palette.length)],

      })
